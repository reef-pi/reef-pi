No-flash first-paint script

Paste into <head> before any CSS — reads localStorage and sets data-theme before render

(function(){
  var t = localStorage.getItem('reefpi.theme') || 'system';
  var r = t === 'system'
    ? (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light')
    : t;
  if (r !== 'light') document.documentElement.setAttribute('data-theme', r);
})();
